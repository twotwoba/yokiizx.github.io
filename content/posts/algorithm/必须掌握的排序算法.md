##### 冒泡排序

```JavaScript
function bubble(arr) {
  for(let i = 0; i < arr.length; i++) {
    let sorted = true
    for(let j = 0; j < arr.length - i; j++) {
      if(arr[j] < arr[j + 1]) {
        swap(j, j + 1)
        sorted = false
      }
    }
    if(sorted) break
  }
}
```
