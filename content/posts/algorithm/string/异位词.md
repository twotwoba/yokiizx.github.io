var groupAnagrams = function(strs) {
    const map = {}
    for(const s of strs) {
        const count = new Array(26).fill(0)
        for(const c of s) {
            count[c.charCodeAt() - 'a'.charCodeAt()]++ // åˆ¶ä½œé”®
        }

};
